plt.figure(1, figsize=(8,8))
xpos = np.arange(28,45)
temps= [1,11,25,44,55,66,77]
profs = [ I[xpos] for I in [I_hura_1, I_hura_11, I_hura_25, I_hura_44, I_hura_55, I_hura_66, I_hura_77]]
from scipy.interpolate import interp1d
fits = []
fits_qs = []
qs_at_max = []
for i,ypos in enumerate(profs):
    Interp = interp1d(xpos, ypos, kind='linear')
    new_xpos = np.linspace( xpos[0], xpos[-1], 500)
    new_ypos = smooth( Interp( new_xpos), beta=60, window_size=60)
    
    fits.append( new_ypos)
    new_qs = np.linspace(q_hura[xpos[0]], q_hura[xpos[-1]], 500)
    fits_qs.append( new_qs)
    plt.plot(new_qs,  new_ypos,lw=3, label="%d C"%temps[i])
    
    idx_at_max = np.argmax(new_ypos)
    q_at_max = new_qs[idx_at_max]
    plt.plot( q_at_max, new_ypos[ idx_at_max], 'ko')
    qs_at_max.append( q_at_max)
    
plt.xlabel("Q inverse angstrom", fontsize=16)
plt.ylabel("hura intensity", fontsize=16)
plt.legend(prop={'size':16}, loc=2)
#plt.legend([ "%d C"%x for x in temps])
plt.gca().tick_params(labelsize=14, length=9)
